\begin{align}
\arg\min_{T, u(t), q(t)} \quad J(T, u(t), q(t)) \quad \text{s.t.}\\
q(0) = q_{start} \quad q(T) = q_{goal}\\
\mathcal{B}(q(t)) \subset \mathcal{W}_{free} \quad \forall t \in [0, T]\\
\dot q(t) = f(q(t), u(t)) \quad \forall t \in [0, T)
\end{align}
$$
